## 创建一千个空文件
```bash
#!/bin/sh
source ~/Downloads/10000-file
    
for n in `seq 1 1000`
do
 touch  ~/Downloads/10000-file/linux-${n}.csv
done
```

## 复制一个文件一千次！
```bash
#!/bin/sh
source ~/Downloads/10000-file
    
for n in `seq 1 1000`
do
 cp -rvf ~/Downloads/10000-file/gbk-csv.csv ~/Downloads/10000-file/${n}.csv
done
```
