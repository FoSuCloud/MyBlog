## 微任务宏任务执行顺序
```
		console.log('1');
			// new Promise是异步微任务,在一个执行循环结束后执行
			new Promise(function(resolve,reject){
				console.log('2');
				setTimeout(function(){
					console.log('5');
					// resolve('success');//此时先打印5再去打印4
				},0);
				resolve('success');//此时先执行then,console.log('4')再去执行5
			}).then(()=>{
				console.log('4')
			})
			console.log('3');
			// setTimeout属于异步宏任务,在一次事件循环之后,如果有微任务先执行微任务,然后再来执行异步宏任务
			setTimeout(function(){
				console.log('6')
			},0)
```
