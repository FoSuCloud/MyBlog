## new构造函数继承
```
			// new 构造函数继承
			function father(){
				this.say=function(){
					console.log(this.name)
				}
			}
			function child(name){
				this.name=name;
				// 如果父类需要参数则要传参
				father.call(this);
			}
			var obj=new child('haha')
			obj.say();// haha
```

## 原型链继承
```
			// 原型链继承
			function father(){
				this.say=function(){
					console.log(this.age)
				}
			}
			function child(age){
				this.age=age;
			}
			child.prototype=new father();
			var obj=new child(33);
			obj.say()
```
