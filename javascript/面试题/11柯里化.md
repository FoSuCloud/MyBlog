## 柯里化
* 柯里化是一种将多个参数的一个函数转换为一系列使用一个参数的函数技术
* 例子：
```javascript
function add(a, b) {
    return a + b;
}

// 执行 add 函数，一次传入两个参数即可
add(1, 2) // 3

// 假设有一个 curry 函数可以做到柯里化
var addCurry = curry(add);
addCurry(1)(2) // 
```

## 柯里化实现
1. * 注意，`函数必须指定形参个数。`
```javascript
function curry(fn,len,...args){
    args = args || []
    let func = function (...inner){
        let _args = [...args,...inner]
        if(_args.length>=len){
            return fn.call(this,..._args)
        }else{
            return curry(fn,len,..._args)
        }
    }
    return func;
}
function add(){
    return Array.from(arguments).reduce((a,b) => a+b)
}
let fn = curry(add,4) // 长度为4
console.log(fn(3)(2,4)(1)) // 10
console.log(fn(3)(2,4)) // 长度不够，打印函数内容体
console.log(fn(3)(2,4,5,6)) // 20
console.log(fn(3)(2,4,5,6)(5)); // 报错，超出长度了！ fn(...)(...) is not a function
```

2. 使用toString方式
* `需要注意的是，toString方法只会在函数需要进行转换为字符串的时候才能自动调用，所以需要String(res(1,2,3,4))`
```javascript
        function curry(fn,...args){
            args = args || [];
            let fun =  function(...innerArgs){
                let _args = args.slice(0); // 深拷贝
                _args.push(...innerArgs);
                return curry.call(this,fn,..._args)  // 返回保存了参数的函数
            }
            fun.toString =  () => {
                return fn.apply(this,args) // 最终执行
            }
            return fun;
        }
        function add(...args){
            return Array.from(args).reduce((a,b)=> a+b);
        }
        let res=curry(add)
        console.log(curry(add,1,2,3,4))
        console.log(res(1,2,3,4)) // 函数的内容体
        console.log(res(1,2,3)(4))// 函数的内容体
        console.log(res(1,2)(3)(4))// 函数的内容体
        console.log(String(res(1,2)(3)(4)))// 10
```
* 为什么不使用valueOf来替代toString?这是因为valueOf一般只在数学运算符和关系运算符作用于数字文本表示的对象时才会自动调用
* 否则一般不会自动调用valueOf方法
```javascript
        function curry(fn,...args){
            args = args || [];
            let fun =  function(...innerArgs){
                let _args = args.slice(0); // 深拷贝
                _args.push(...innerArgs);
                return curry.call(this,fn,..._args)  // 返回保存了参数的函数
            }
            // 只有在使用数学运算符和关系运算符针对数字文本的时候才会自动调用valueOf
            fun.valueOf =  () => {
                return fn.apply(this,args) // 最终执行
            }
            // 否则自动调用的是toString来获取文本值
            fun.toString = ()=>{
                return 'toString'
            }
            return fun;
        }
        function add(...args){
            return Array.from(args).reduce((a,b)=> a+b);
        }
        let res=curry(add)
        console.log(+res(1,2,3,4)) // 10
        console.log(res(1,2,3)(4) >= 10) // true
        console.log(res(1,2)(3)(4)) // ƒ toString
```

## 不封装实现柯里化
```javascript
        function sum(){
           let context=this;
           let arr=Array.from(arguments) || [];
           let fn=function (...innerArgs){
               arr.push(...innerArgs);
               return sum.apply(context,arr)
           }
           fn.toString = function (){
               return arr.reduce((a,b)=> a+b)
           }
           return fn;
       }
       console.log(sum(1,2))
       console.log(sum(1,2)(3))
       console.log(sum(1,2)(3,4))
       console.log(sum(1,2)(3,4)(5))
```
