## 冒泡排序

#### 75. 颜色分类
```js
/**
 Do not return anything, modify nums in-place instead.
 */
function sortColors(nums: number[]): void {
    if(nums.length<=1){
        return;
    }
    for(let i=0;i<nums.length-1;i++){
        for(let j=1;j<nums.length-i;j++){
            if(nums[j-1]>nums[j]){
                [nums[j-1],nums[j]]=[nums[j],nums[j-1]];
            }
        }
    }
};
```
