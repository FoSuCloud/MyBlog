## 二叉树的最大深度
* [https://leetcode-cn.com/problems/maximum-depth-of-binary-tree/submissions/]()
```javascript
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function(root) {
    if(!root){
        return 0
    }
    let max=0;
    function dfs(tree,num){
        let {left,right}=tree;
        num++;
        max=Math.max(num,max)
        if(left){
            dfs(left,num)
        }
        if(right){
            dfs(right,num)
        }
    }
    dfs(root,0)
    return max;
};
```

## 路径总和
* [https://leetcode-cn.com/problems/path-sum/]()
```javascript
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} targetSum
 * @return {boolean}
 */
var hasPathSum = function(root, targetSum) {
    if(!root){
        return false;
    }
    let res=false;
    function dfs(tree,sum){
        if(res){
            return;
        }
        if(!tree){
            return;
        }
        let {left,val,right}=tree;
        sum+=val;
        if(sum===targetSum){
            // 叶子节点
            if(!tree.left && !tree.right){
                res=true;
                return;
            }
        }
        if(left){
            dfs(tree.left,sum)
        }
        if(right){
            dfs(tree.right,sum)
        }
    }
    dfs(root,0)
    return res;
};
```

## 路径总和2
* [https://leetcode-cn.com/problems/path-sum-ii/submissions/]()
```javascript
/**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @param {number} targetSum
 * @return {number[][]}
 */
var pathSum = function(root, targetSum) {
    let res=[]
    if(!root){
        return res;
    }
    function dfs(tree,num,nodes){
        let {left,val,right}=tree;
        num+=val;
        let tmp=nodes.slice()
        
        if(num === targetSum){
            if(!tree.left && !tree.right){
                res.push(nodes.concat(val))
                return;
            }
        }

        if(left){
            dfs(left,num,tmp.concat(val))
        }
        if(right){
            dfs(right,num,tmp.concat(val))
        }
    }
    dfs(root,0,[])
    return res;
};
```

## 岛屿的最大面积
* [https://leetcode-cn.com/problems/max-area-of-island/submissions/]()
```javascript
/**
 * @param {number[][]} grid
 * @return {number}
 */
var maxAreaOfIsland = function(grid) {
    let tmp=0;
    let max=0;
    for(let i=0;i<grid.length;i++){
        for(let j=0;j<grid[0].length;j++){
            if(grid[i][j]===1){
                dfs(i,j)
                max=Math.max(tmp,max)
                tmp=0;
            }
        }
    }
    // dfs
    function dfs(i,j){
        tmp++; // 数量+1
        grid[i][j]=0;
        if(i+1<grid.length && grid[i+1][j]){
            dfs(i+1,j)
        }
        if(i-1>=0 && grid[i-1][j]){
            dfs(i-1,j)
        }
        if(j+1<grid[0].length && grid[i][j+1]){
            dfs(i,j+1)
        }
        if(j-1>=0 && grid[i][j-1]){
            dfs(i,j-1)
        }
    }
    return max;
};
```
