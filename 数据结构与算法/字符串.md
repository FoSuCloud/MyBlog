## 去重后依旧字典序
* 给一个全是小写字母的字符串str，删除多余字符，使得每种字符只保留一个，并且让最终结果字符串字典序最小。
```
dbcacbca =>dabc
acbc => abc

var line=readline().split('');
var flag={index:0,value:line[0]};
var len=line.length;
var arr=[line[0]]
for(var i=len-1;i>0;i--){
    if(arr.indexOf(line[i])!=-1){
        line.splice(i,1)
    }else if((line.indexOf(line[i])!=i)&&(line[i]<flag.value)&&(i>flag.index)){
        line.splice(i,1);
    }else{
        arr.push(line[i])
    }
    flag.value=line[i]
    flag.index=i;
}
console.log(line.join(''))
```

## 寻找最长重复子串

## 字符串归一化
* 通过键盘输入一串小写字母(a~z)组成的字符串。
* 请编写一个字符串归一化程序，统计字符串中相同字符出现的次数，并按字典序输出字符及其出现次数。
* 例如字符串"babcc"归一化后为"a1b2c2"
```
var arr=readline().split('');
var len=arr.length;
var mymap={}
for(var i=0;i<len;i++){
    if(mymap[arr[i]]){
        mymap[arr[i]]++;
    }else{
        mymap[arr[i]]=1;
    }
}
var res=Object.keys(mymap).sort();
var str='';
for(var k in res){
    str=str+res[k]+mymap[res[k]]
}
console.log(str)
```
* `关键在于通过一个对象存储个数，然后通过Object.keys(obj).sort()对对象的属性进行排序`
* `然后再对拍完序的属性数组进行遍历，根据字符串拼接进行输出`

## 最长回文子串
* 中心拓展法
```javascript
var longestPalindrome = function(s) {
            if(s.length<=1){
                return s;
            }
            let str=s[0];
            for(let i=0;i<s.length;i++){
                dp(i,i) // 奇数
                dp(i,i+1) // 偶数
            }
            function dp(left,right){
                while(true){
                    if(left<0 || right>=s.length || s[left]!==s[right]){
                        break;
                    }
                    let tmp=s.slice(left,right+1)
                    str=str.length>=tmp.length?str:tmp;
                    left--;
                    right++;
                }
            }
            return str;
        };
```

## 分割回文串
```javascript
/**
 * @param {string} s
 * @return {string[][]}
 */
var partition = function(s) {
    let res=[];
    // 动态规划
    function dfs(arr,l){
        // 走到终点代表是回文结果
        if(l===s.length){
            res.push(arr);
            return;
        }
        for(let i=l;i<s.length;i++){
            if(isPali(s,l,i)){
                let tmp=arr.slice();
                tmp.push(s.substring(l,i+1));
                // 传递当前的回文子串
                dfs(tmp,i+1);
            }
        }
    }
    // 判断是否是回文字符串
    function isPali(str,l,r){
        while(l<r){
            if(s[l]!==s[r]){
                return false;
            }
            l++;
            r--;
        }
        return true;
    }
    dfs([],0)
    return res;
};
```

## 无重复字符的最长子串
```javascript
/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function(s) {
    let arr=[];
    let max=0;
    for(let i=0;i<s.length;i++){
        let index=arr.indexOf(s[i])
        if(index === -1){
            arr.push(s[i])
        }else{
            max=Math.max(max,arr.length)
            arr.splice(0,index+1)
            arr.push(s[i])
        }
    }
    return Math.max(max,arr.length);
};
```
* 另一种解法，map结合双指针
```javascript
var lengthOfLongestSubstring = function(s) {
    let map=new Map();
    let max=0;
    let l=0;
    for(let r=0;r<s.length;r++){
        if(map.has(s[r])){
            // 注意，没有删除map的值
            l=Math.max(map.get(s[r])+1,l); // 如果map对照的值在指针前面，那就不要改变指针方向
        }
        map.set(s[r],r)
        max=Math.max(max,r-l+1)
    }
    return max;
};
```

## 字符的最短距离
* 中心拓展法
```javascript
/**
 * @param {string} s
 * @param {character} c
 * @return {number[]}
 */
var shortestToChar = function(s, c) {
    let res=[];
    for(let i=0;i<s.length;i++){
        res.push(fn(i))
    }
    function fn(i){
        let l=i,r=i;
        while(l>=0 || r<s.length){
            if(l>=0 && s[l]===c){
                return i-l;
            }
            if(r<s.length && s[r]===c){
                return r-i;
            }
            r++;
            l--;
        }
        return 0;
    }
    return res;
};
```

```javascript
var validPalindrome = function(s) {
    let l=0,r=s.length-1;
    while(l<r){
        if(s[l]!==s[r]){
            let res1=dp(s.substring(0,l)+s.substring(l+1))
            if(res1){
                return true;
            }
            let res2=dp(s.substring(0,r)+s.substring(r+1))
            if(res2){
                return true;
            }
            return false;
        }
        l++;
        r--;
    }
    return true;

        while(l<=r){
            if(str[l]!==str[r]){
                return false;
            }
            l++;
            r--;
        }
        return true;
    }
};
```
